## Fundamentals

### Spark Architecture
```
Driver Program
├── SparkContext
├── DAG Scheduler → TaskScheduler
└── Cluster Manager
    ├── Executor 1 (Worker Node)
    │   ├── Task 1 → Cache/Storage
    │   └── Task 2
    └── Executor 2 (Worker Node)
        ├── Task 3
        └── Task 4
```

**Key Points:**
- Driver: Single point of failure, holds SparkContext
- Executors: JVM processes, execute tasks in parallel
- DAG: Directed Acyclic Graph of RDD transformations
- Stages: Bounded by shuffle operations (wide transformations)

### RDD Lineage vs DataFrame Execution Plan
```python
# RDD Lineage (physical execution path)
rdd = sc.textFile("data.txt")
words = rdd.flatMap(lambda line: line.split())
word_counts = words.map(lambda word: (word, 1)).reduceByKey(lambda a, b: a + b)
print(word_counts.toDebugString())  # Shows lineage graph

# DataFrame Execution Plan (logical + physical)
df = spark.read.text("data.txt")
word_counts = df.select(explode(split(col("value"), " ")).alias("word")) \
                .groupBy("word").count()
word_counts.explain(extended=True)  # Shows all plans
```

### Lazy Evaluation Deep Dive
**Transformations** (lazy):
- `map`, `filter`, `flatMap`, `union`, `join`, `groupBy`

**Actions** (eager):
- `collect`, `count`, `take`, `save`, `foreach`
- Trigger job execution

### RDDs - Advanced Operations

#### Transformations
```python
# mapPartitions - Process entire partitions
def process_partition(iterator):
    # Expensive setup once per partition
    model = load_ml_model()
    for record in iterator:
        yield model.predict(record)

rdd.mapPartitions(process_partition)

# treeReduce - Hierarchical reduction (O(log n) depth)
rdd.treeReduce(lambda x, y: x + y, depth=3)

# Custom partitioner
class CustomPartitioner:
    def __init__(self, num_partitions):
        self.num_partitions = num_partitions
    
    def getPartition(self, key):
        return hash(key) % self.num_partitions

rdd.partitionBy(CustomPartitioner(10))
```

#### Actions
```python
# fold - With zero value
rdd.fold(0, lambda acc, x: acc + x)

# aggregate - Different types for accumulator and combiner
rdd.aggregate(
    (0, 0),  # Zero value: (sum, count)
    lambda acc, x: (acc[0] + x, acc[1] + 1),  # Seq op
    lambda acc1, acc2: (acc1[0] + acc2[0], acc1[1] + acc2[1])  # Comb op
)
```

#### Partitioning Strategies
```python
# Range partitioning (for sorted data)
rdd.partitionBy(RangePartitioner(numPartitions=10, rdd))

# Hash partitioning (default)
rdd.partitionBy(HashPartitioner(numPartitions=10))

# Custom partitioning for skewed data
class SkewPartitioner:
    def getPartition(self, key):
        if key in high_frequency_keys:
            return hash(key) % (num_partitions // 2)
        return (num_partitions // 2) + (hash(key) % (num_partitions // 2))
```

### Join Strategies

#### Broadcast Hash Join
```python
# Automatic broadcast (< spark.sql.autoBroadcastJoinThreshold = 10MB)
small_df.join(large_df, "key")  # Spark decides

# Explicit broadcast
from pyspark.sql.functions import broadcast
large_df.join(broadcast(small_df), "key")

# Configuration

```

#### Sort Merge Join
```python
# Default for large tables
# Both sides sorted and partitioned by join key
df1.join(df2, "key")  # Uses SMJ if both sides are large

# Optimize by pre-sorting
df1_sorted = df1.sortWithinPartitions("key")
df2_sorted = df2.sortWithinPartitions("key")
result = df1_sorted.join(df2_sorted, "key")
```

#### Bucket Join
```python
# Pre-partition and sort data during write
df1.write \
   .bucketBy(10, "key") \
   .sortBy("key") \
   .saveAsTable("bucketed_table1")

df2.write \
   .bucketBy(10, "key") \
   .sortBy("key") \
   .saveAsTable("bucketed_table2")

# Join without shuffle
bucketed_df1 = spark.table("bucketed_table1")
bucketed_df2 = spark.table("bucketed_table2")
result = bucketed_df1.join(bucketed_df2, "key")  # No shuffle!
```

### Data Skew Solutions

#### Salting Technique
```python
import random
from pyspark.sql.functions import rand, lit, array, explode

# Detect skew
key_counts = df.groupBy("key").count()
skewed_keys = key_counts.filter(col("count") > threshold).select("key")

# Salt skewed keys
SALT_BUCKETS = 100

def add_salt(df, join_keys, salt_buckets):
    return df.withColumn("salt", (rand() * salt_buckets).cast("int")) \
             .withColumn("salted_key", concat_ws("_", col("key"), col("salt")))

# For dimension table, duplicate with all salt values
def replicate_with_salt(df, salt_buckets):
    salts = [lit(i) for i in range(salt_buckets)]
    return df.withColumn("salt", explode(array(*salts))) \
             .withColumn("salted_key", concat_ws("_", col("key"), col("salt")))

# Apply salting
salted_fact = add_salt(fact_df, ["key"], SALT_BUCKETS)
replicated_dim = replicate_with_salt(dim_df, SALT_BUCKETS)
result = salted_fact.join(replicated_dim, "salted_key")
```

#### Isolated Broadcast Map
```python
# Handle mixed skew scenario
def skew_join(large_df, small_df, join_key, skew_threshold=1000):
    # Identify skewed keys
    key_counts = large_df.groupBy(join_key).count()
    skewed_keys = key_counts.filter(col("count") > skew_threshold) \
                           .select(join_key).rdd.map(lambda x: x[0]).collect()
    
    # Split large DataFrame
    skewed_large = large_df.filter(col(join_key).isin(skewed_keys))
    normal_large = large_df.filter(~col(join_key).isin(skewed_keys))
    
    # Split small DataFrame
    skewed_small = small_df.filter(col(join_key).isin(skewed_keys))
    normal_small = small_df.filter(~col(join_key).isin(skewed_keys))
    
    # Normal join with broadcast
    normal_result = normal_large.join(broadcast(normal_small), join_key)
    
    # Skewed join with salting
    salted_result = add_salt(skewed_large, [join_key], 100) \
                   .join(replicate_with_salt(skewed_small, 100), "salted_key")
    
    return normal_result.unionByName(salted_result.drop("salt", "salted_key"))
```

#### Watermarking Strategy
```python

    .withWatermark("event_time", "1 hour")  # Allow 1 hour late data

# Events older than this are dropped

# Monitor watermark
def monitor_watermark(batch_df, batch_id):
```

#### Stateful Operations
```python
from pyspark.sql.streaming.state import GroupState, GroupStateTimeout

# mapGroupsWithState for custom stateful processing
def update_user_session(key, values, state: GroupState):
    if state.hasTimedOut:
        # Handle timeout
        final_session = state.get()
        state.remove()
        return final_session
    
    # Update state with new events
    if state.exists:
        current_session = state.get()
    else:
        current_session = {"user_id": key[0], "events": [], "start_time": None}
    
    for event in values:
        current_session["events"].append(event)
        if not current_session["start_time"]:
            current_session["start_time"] = event.timestamp
    
    # Set timeout (30 minutes of inactivity)
    state.setTimeoutDuration("30 minutes")
    state.update(current_session)
    
    return current_session

# Apply stateful transformation
user_sessions = events_df \
    .groupBy("user_id") \
    .mapGroupsWithState(
        outputMode=OutputMode.Update(),
        timeoutConf=GroupStateTimeout.ProcessingTimeTimeout
    )(update_user_session)
```

### Distributed Model Training
```python
from pyspark.ml.classification import RandomForestClassifier
from pyspark.ml.evaluation import BinaryClassificationEvaluator
from pyspark.ml.tuning import CrossValidator, ParamGridBuilder

# Stratified sampling for imbalanced datasets
def stratified_split(df, label_col, train_ratio=0.8, seed=42):
    fractions = df.select(label_col).distinct().rdd.map(lambda x: (x[0], train_ratio)).collectAsMap()
    train_df = df.sampleBy(label_col, fractions, seed)
    test_df = df.subtract(train_df)
    return train_df, test_df

train_df, test_df = stratified_split(df, "label")

# Hyperparameter tuning
rf = RandomForestClassifier(featuresCol="features", labelCol="label")
paramGrid = ParamGridBuilder() \
    .addGrid(rf.numTrees, [10, 50, 100]) \
    .addGrid(rf.maxDepth, [5, 10, 15]) \
    .build()

evaluator = BinaryClassificationEvaluator()
crossval = CrossValidator(
    estimator=rf,
    estimatorParamMaps=paramGrid,
    evaluator=evaluator,
    numFolds=5
)

cv_model = crossval.fit(train_df)
```

## Interview Question Patterns

### Pattern 1: Data Processing & Aggregation
```python
# Common Question: "Find top N customers by revenue per month"
def top_customers_by_month(orders_df, n=10):
    monthly_revenue = orders_df \
        .withColumn("year_month", date_format(col("order_date"), "yyyy-MM")) \
        .groupBy("year_month", "customer_id") \
        .agg(sum("order_amount").alias("monthly_revenue")) \
        .withColumn("rank", 
                   row_number().over(
                       Window.partitionBy("year_month")
                             .orderBy(desc("monthly_revenue"))
                   )) \
        .filter(col("rank") <= n)
    
    return monthly_revenue

# Advanced: Handle data skew with salting
def top_customers_skew_resistant(orders_df, n=10):
    # Add random salt for initial aggregation
    salted_df = orders_df \
        .withColumn("salt", (rand() * 100).cast("int")) \
        .withColumn("salted_customer", concat(col("customer_id"), lit("_"), col("salt")))
    
    # First level aggregation
    partial_agg = salted_df \
        .groupBy("year_month", "salted_customer", "customer_id") \
        .agg(sum("order_amount").alias("partial_revenue"))
    
    # Final aggregation
    final_agg = partial_agg \
        .groupBy("year_month", "customer_id") \
        .agg(sum("partial_revenue").alias("monthly_revenue"))
    
    return final_agg.withColumn("rank", 
                               row_number().over(
                                   Window.partitionBy("year_month")
                                         .orderBy(desc("monthly_revenue"))
                               )).filter(col("rank") <= n)
```

### Pattern 2: Complex Joins & Deduplication
```python
# Question: "Join user events with user profiles, handle duplicates"
def clean_user_events_join(events_df, profiles_df):
    # Deduplicate events (keep latest by timestamp)
    events_clean = events_df \
        .withColumn("rank", 
                   row_number().over(
                       Window.partitionBy("user_id", "event_type")
                             .orderBy(desc("timestamp"))
                   )) \
        .filter(col("rank") == 1) \
        .drop("rank")
    
    # Deduplicate profiles (keep latest version)
    profiles_clean = profiles_df \
        .withColumn("rank",
                   row_number().over(
                       Window.partitionBy("user_id")
                             .orderBy(desc("last_updated"))
                   )) \
        .filter(col("rank") == 1) \
        .drop("rank")
    
    # Efficient join strategy
    if profiles_clean.count() < 10000:  # Broadcast small table
        result = events_clean.join(broadcast(profiles_clean), "user_id", "left")
    else:  # Sort-merge join
        result = events_clean.sortWithinPartitions("user_id") \
                           .join(profiles_clean.sortWithinPartitions("user_id"), 
                                 "user_id", "left")
    
    return result
```

```python
# Question: "Calculate moving averages and detect anomalies"
def time_series_analysis(metrics_df, window_days=7, anomaly_threshold=2.0):
    # Calculate moving statistics
    window = Window.partitionBy("metric_name") \
                   .orderBy("date") \
                   .rowsBetween(-(window_days-1), 0)
    
    stats_df = metrics_df \
        .withColumn("moving_avg", avg("value").over(window)) \
        .withColumn("moving_std", 
                   sqrt(avg(pow(col("value") - avg("value").over(window), 2)).over(window))) \
        .withColumn("z_score", 
                   (col("value") - col("moving_avg")) / col("moving_std"))
    
    # Detect anomalies
    anomalies = stats_df \
        .withColumn("is_anomaly", abs(col("z_score")) > anomaly_threshold) \
        .withColumn("anomaly_type",
                   when(col("z_score") > anomaly_threshold, "high")
                   .when(col("z_score") < -anomaly_threshold, "low")
                   .otherwise("normal"))
    
    return anomalies
```

```python
# Question: "Find shortest paths in a network"
def shortest_paths_bfs(edges_df, source_id, max_iterations=10):
    # Initialize distances
    vertices_df = edges_df.select("src").distinct().withColumnRenamed("src", "id") \
                         .union(edges_df.select("dst").distinct().withColumnRenamed("dst", "id")) \
                         .distinct() \
                         .withColumn("distance", 
                                   when(col("id") == source_id, 0).otherwise(float('inf')))
    
    # Iterative BFS
    for i in range(max_iterations):
        # Join vertices with edges
        updated = vertices_df.alias("v").join(edges_df.alias("e"), col("v.id") == col("e.src")) \
                            .withColumn("new_distance", col("v.distance") + col("e.weight")) \
                            .select(col("e.dst").alias("id"), "new_distance")
        
        # Update distances
        old_vertices = vertices_df.join(updated, "id", "left_anti")
        new_distances = vertices_df.join(updated, "id") \
                                 .withColumn("distance", 
                                           least(col("distance"), col("new_distance"))) \
                                 .select("id", "distance")
        
        vertices_df = old_vertices.union(new_distances)
        
        # Check for convergence
        if updated.count() == 0:
            break
    
    return vertices_df.filter(col("distance") != float('inf'))
```

### Pattern 5: Real-time Stream Processing
```python
# Question: "Implement session-based aggregations"
def session_aggregation(events_stream, session_timeout_minutes=30):
    # Define state schema
    from pyspark.sql.types import MapType
    import time
    
    def update_session_state(key, values, state):
        session_timeout = session_timeout_minutes * 60 * 1000  # milliseconds
        current_time = int(time.time() * 1000)
        
        if state.hasTimedOut:
            # Session expired, emit final result
            if state.exists:
                final_session = state.get()
                state.remove()
                return [final_session]
            return []
        
        # Process new events
        session_data = state.get() if state.exists else {
            "user_id": key,
            "events": [],
            "session_start": None,
            "last_activity": None
        }
        
        for event in values:
            event_time = int(event.timestamp.timestamp() * 1000)
            
            # Check if new session needed
            if (session_data["last_activity"] and 
                event_time - session_data["last_activity"] > session_timeout):
                # Emit previous session and start new one
                old_session = session_data.copy()
                state.remove()
                
                # Start new session
                session_data = {
                    "user_id": key,
                    "events": [event],
                    "session_start": event_time,
                    "last_activity": event_time
                }
                state.update(session_data)
                state.setTimeoutDuration(f"{session_timeout_minutes} minutes")
                
                return [old_session]
            else:
                # Continue current session
                session_data["events"].append(event)
                if not session_data["session_start"]:
                    session_data["session_start"] = event_time
                session_data["last_activity"] = event_time
        
        state.update(session_data)
        state.setTimeoutDuration(f"{session_timeout_minutes} minutes")
        return []
    
    # Apply session aggregation
    sessions = events_stream \
        .groupBy("user_id") \
        .mapGroupsWithState(
            outputMode=OutputMode.Append(),
            timeoutConf=GroupStateTimeout.ProcessingTimeTimeout
        )(update_session_state)
    
    return sessions
```
